1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "LoadBalancer" .Values.nginx.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "kappi1.fullname" . }}-nginx'
{{- else if contains "ClusterIP" .Values.nginx.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "kappi1.fullname" . }}-nginx)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- end }}

2. Check the status of your deployment:
  kubectl get pods -l app.kubernetes.io/name={{ include "kappi1.name" . }}

3. View logs:
  Backend:  kubectl logs -l app.kubernetes.io/component=backend
  Frontend: kubectl logs -l app.kubernetes.io/component=frontend
  Nginx:    kubectl logs -l app.kubernetes.io/component=nginx

4. Access the API documentation:
  kubectl port-forward svc/{{ include "kappi1.fullname" . }}-backend 8001:8001
  Then open: http://localhost:8001/docs

{{- if .Values.mongodb.enabled }}
5. MongoDB is running. Connect using:
  kubectl port-forward svc/{{ include "kappi1.fullname" . }}-mongodb 27017:27017
{{- end }}
